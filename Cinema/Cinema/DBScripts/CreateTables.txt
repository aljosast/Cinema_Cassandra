 CREATE TABLE "Filmovi" (     "ID" text,     "Naziv" text,     "DugiOpis" text,     
 "Opis" text,     "Reziser" text,     "Slika" text,     "Zanr" text,     PRIMARY KEY ("ID") );

 CREATE TABLE "Glumci" (     "FilmID" text,     "Ime" text,     "TipUloge" text,     
 "Uloga" text,    PRIMARY KEY ("FilmID", "Ime") );

 CREATE TABLE "Bioskop" (     "ID" text,     "Naziv" text,     "Grad" text,     
 "Adresa" text,    PRIMARY KEY ("Grad", "ID") );

  CREATE TABLE "Projekcija" (     "ID" text,     "FilmID" text,     "BioskopID" text,
 "BrojSale" int,     "BrojMesta" int,      "BrojRezervacija" int,      "Vreme" timestamp,
 "Naziv" text,     "Slika" text,    PRIMARY KEY ("BioskopID", "ID") );

  CREATE TABLE "Rezervacija" (     "ID" text,     "Username" text,     "ProjekcijaID" text,
  "VremeProjekcije" timestamp,     "NazivFilma" text,     "NazivBioskopa" text,
  "AdresaBioskopa" text,    PRIMARY KEY ("Username", "ID") );

  CREATE TABLE "Korisnik" (     "Username" text,     "Ime" text,     "Prezime" text,     
 "Godine" int,    "Password" text,     "Role" text,    PRIMARY KEY ("Username") );
